  App.js: |
    import Widget from "@meursyphus/flitter-react";
    import { SankeyChart } from "@meursyphus/headless-chart";
    import {
      Text,
      Container,
      EdgeInsets,
      TextStyle,
      Column,
      SizedBox,
      Center,
    } from "@meursyphus/flitter";

    const data = {
      nodes: [
        { id: "solar", label: "Solar", color: "#f28e2b" },
        { id: "wind", label: "Wind", color: "#4e79a7" },
        { id: "hydro", label: "Hydro", color: "#76b7b2" },
        { id: "coal", label: "Coal", color: "#9c755f" },
        { id: "grid", label: "Power Grid", color: "#bab0ac" },
        { id: "residential", label: "Residential", color: "#59a14f" },
        { id: "commercial", label: "Commercial", color: "#edc948" },
        { id: "industrial", label: "Industrial", color: "#e15759" },
      ],
      links: [
        { source: "solar", target: "grid", value: 120 },
        { source: "wind", target: "grid", value: 80 },
        { source: "hydro", target: "grid", value: 60 },
        { source: "coal", target: "grid", value: 200 },
        { source: "grid", target: "residential", value: 180 },
        { source: "grid", target: "commercial", value: 150 },
        { source: "grid", target: "industrial", value: 130 },
      ],
      title: "Energy Flow",
    };

    const chart = SankeyChart({
      data,
      custom: {
        layout: ({ title, sankey }) =>
          Container({
            padding: EdgeInsets.all(24),
            child: Column({
              children: [
                Center({
                  child: Text("Energy Flow Diagram (TWh)", {
                    style: new TextStyle({
                      fontSize: 16,
                      fontWeight: "bold",
                      color: "#333333",
                      fontFamily: "Noto Sans JP",
                    }),
                  }),
                }),
                SizedBox({ height: 20 }),
                sankey,
              ],
            }),
          }),
        nodeLabel: ({ label }) =>
          Text(label, {
            style: new TextStyle({
              fontSize: 12,
              color: "#444444",
              fontWeight: "bold",
              fontFamily: "Noto Sans JP",
            }),
          }),
      },
    });

    export default function App() {
      return (
        <Widget
          width="auto"
          height="450px"
          widget={chart}
          renderer="svg"
        />
      );
    }
---

# Sankey Chart Example

A sankey diagram illustrating energy flow from sources (Solar, Wind, Hydro, Coal) through the power grid to consumption sectors (Residential, Commercial, Industrial).
