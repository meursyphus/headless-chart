  App.js: |
    import Widget from "@meursyphus/flitter-react";
    import { StackedAreaChart } from "@meursyphus/headless-chart";
    import {
      Text,
      Container,
      EdgeInsets,
      TextStyle,
      Column,
      Row,
      SizedBox,
      MainAxisAlignment,
      MainAxisSize,
      BoxDecoration,
      Padding,
      Center,
    } from "@meursyphus/flitter";

    const colors = ["#4e79a7", "#f28e2b", "#e15759", "#76b7b2"];

    const data = {
      labels: ["Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"],
      datasets: [
        { legend: "Direct", values: [30, 35, 40, 38, 45, 50, 48, 52, 55, 60, 58, 65] },
        { legend: "Organic Search", values: [50, 55, 60, 65, 70, 68, 72, 78, 82, 85, 88, 92] },
        { legend: "Social Media", values: [20, 25, 28, 32, 30, 35, 40, 38, 42, 45, 48, 52] },
        { legend: "Referral", values: [10, 12, 15, 14, 18, 20, 22, 25, 24, 28, 30, 32] },
      ],
    };

    const chart = StackedAreaChart({
      data,
      custom: {
        layout: ({ title, legends, plot }) =>
          Container({
            padding: EdgeInsets.only({ left: 50, bottom: 50, right: 20, top: 30 }),
            child: Column({
              children: [
                Center({
                  child: Text("Website Traffic Sources (2024)", {
                    style: new TextStyle({
                      fontSize: 16,
                      fontWeight: "bold",
                      color: "#333333",
                      fontFamily: "Noto Sans JP",
                    }),
                  }),
                }),
                SizedBox({ height: 8 }),
                Row({
                  mainAxisAlignment: MainAxisAlignment.center,
                  children: legends,
                }),
                SizedBox({ height: 12 }),
                plot,
              ],
            }),
          }),
        legend: ({ name, index }) =>
          Padding({
            padding: EdgeInsets.symmetric({ horizontal: 6 }),
            child: Row({
              mainAxisSize: MainAxisSize.min,
              children: [
                Container({
                  width: 12,
                  height: 12,
                  decoration: new BoxDecoration({
                    color: colors[index % colors.length],
                    borderRadius: 2,
                  }),
                }),
                SizedBox({ width: 5 }),
                Text(name, {
                  style: new TextStyle({
                    fontSize: 12,
                    color: "#666666",
                    fontFamily: "Noto Sans JP",
                  }),
                }),
              ],
            }),
          }),
        xAxisLabel: ({ name }) =>
          Padding({
            padding: EdgeInsets.only({ top: 4 }),
            child: Text(name, {
              style: new TextStyle({
                fontSize: 11,
                color: "#666666",
                fontFamily: "Noto Sans JP",
              }),
            }),
          }),
        yAxisLabel: ({ name }) =>
          Padding({
            padding: EdgeInsets.only({ right: 4 }),
            child: Text(name, {
              style: new TextStyle({
                fontSize: 11,
                color: "#666666",
                fontFamily: "Noto Sans JP",
              }),
            }),
          }),
        xAxisTick: () => Container({ height: 6, width: 1, color: "#DDDDDD" }),
        yAxisTick: () => Container({ height: 1, width: 6, color: "#DDDDDD" }),
        xAxisLine: () => Container({ color: "#BBBBBB", height: 1, width: Infinity }),
        yAxisLine: () => Container({ color: "#BBBBBB", width: 1, height: Infinity }),
        gridXLine: () => Container({ height: 1, color: "#EEEEEE" }),
        gridYLine: () => Container({ width: 1, color: "#EEEEEE" }),
      },
    });

    export default function App() {
      return (
        <Widget
          width="auto"
          height="450px"
          widget={chart}
          renderer="svg"
        />
      );
    }

# Stacked Area Chart Example

A stacked area chart visualizing website traffic sources over 12 months, showing cumulative trends for Direct, Organic Search, Social Media, and Referral channels.
