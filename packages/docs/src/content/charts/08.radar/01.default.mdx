  App.js: |
    import Widget from "@meursyphus/flitter-react";
    import { RadarChart } from "@meursyphus/headless-chart";
    import {
      Text,
      Container,
      EdgeInsets,
      TextStyle,
      Column,
      Row,
      SizedBox,
      MainAxisAlignment,
      MainAxisSize,
      BoxDecoration,
      Padding,
      Center,
    } from "@meursyphus/flitter";

    const colors = ["rgba(78, 121, 167, 0.4)", "rgba(242, 142, 43, 0.4)"];
    const borderColors = ["#4e79a7", "#f28e2b"];

    const data = {
      labels: ["Attack", "Defense", "Speed", "Stamina", "Technique", "Power"],
      datasets: [
        {
          legend: "Player A",
          values: [85, 72, 90, 65, 80, 78],
        },
        {
          legend: "Player B",
          values: [70, 88, 60, 82, 75, 92],
        },
      ],
      title: "Player Comparison",
    };

    const chart = RadarChart({
      data,
      custom: {
        layout: ({ title, legends, radar }) =>
          Container({
            padding: EdgeInsets.all(24),
            child: Column({
              children: [
                Center({
                  child: Text("Player Comparison", {
                    style: new TextStyle({
                      fontSize: 18,
                      fontWeight: "bold",
                      color: "#333333",
                      fontFamily: "Noto Sans JP",
                    }),
                  }),
                }),
                SizedBox({ height: 8 }),
                Row({
                  mainAxisAlignment: MainAxisAlignment.center,
                  children: legends,
                }),
                SizedBox({ height: 8 }),
                radar,
              ],
            }),
          }),
        legend: ({ name, index }) =>
          Padding({
            padding: EdgeInsets.symmetric({ horizontal: 8 }),
            child: Row({
              mainAxisSize: MainAxisSize.min,
              children: [
                Container({
                  width: 14,
                  height: 14,
                  decoration: new BoxDecoration({
                    color: colors[index % colors.length],
                    border: BoxDecoration.border
                      ? undefined
                      : undefined,
                    borderRadius: 3,
                  }),
                }),
                SizedBox({ width: 6 }),
                Text(name, {
                  style: new TextStyle({
                    fontSize: 13,
                    color: "#555555",
                    fontFamily: "Noto Sans JP",
                  }),
                }),
              ],
            }),
          }),
        axisLabel: ({ label }) =>
          Text(label, {
            style: new TextStyle({
              fontSize: 12,
              color: "#555555",
              fontFamily: "Noto Sans JP",
              fontWeight: "bold",
            }),
          }),
      },
    });

    export default function App() {
      return (
        <Widget
          width="auto"
          height="500px"
          widget={chart}
          renderer="svg"
        />
      );
    }
---
