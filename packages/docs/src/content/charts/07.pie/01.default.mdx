  App.js: |
    import Widget from "@meursyphus/flitter-react";
    import { PieChart } from "@meursyphus/headless-chart";
    import {
      Text,
      Container,
      EdgeInsets,
      TextStyle,
      Column,
      Row,
      SizedBox,
      MainAxisAlignment,
      MainAxisSize,
      BoxDecoration,
      Padding,
      Center,
    } from "@meursyphus/flitter";

    const colors = [
      "#4e79a7", "#f28e2b", "#e15759", "#76b7b2",
      "#59a14f", "#edc948", "#b07aa1", "#ff9da7",
    ];

    const data = {
      labels: ["Chrome", "Safari", "Firefox", "Edge", "Opera", "Samsung", "Other"],
      values: [64.87, 18.78, 3.16, 5.43, 2.37, 2.56, 2.83],
      title: "Browser Market Share",
    };

    const chart = PieChart({
      data,
      colors,
      custom: {
        layout: ({ title, pie, legends }) =>
          Container({
            padding: EdgeInsets.all(24),
            child: Column({
              children: [
                title,
                SizedBox({ height: 16 }),
                pie,
                SizedBox({ height: 16 }),
                Row({
                  mainAxisAlignment: MainAxisAlignment.center,
                  children: legends,
                }),
              ],
            }),
          }),
        legend: ({ name, index }) =>
          Padding({
            padding: EdgeInsets.symmetric({ horizontal: 6 }),
            child: Row({
              mainAxisSize: MainAxisSize.min,
              children: [
                Container({
                  width: 12,
                  height: 12,
                  decoration: new BoxDecoration({
                    color: colors[index % colors.length],
                    borderRadius: 3,
                  }),
                }),
                SizedBox({ width: 4 }),
                Text(name, {
                  style: new TextStyle({
                    fontSize: 11,
                    color: "#666666",
                    fontFamily: "Noto Sans JP",
                  }),
                }),
              ],
            }),
          }),
        title: ({ name }) =>
          Center({
            child: Text(name, {
              style: new TextStyle({
                fontSize: 18,
                fontWeight: "bold",
                color: "#333333",
                fontFamily: "Noto Sans JP",
              }),
            }),
          }),
        dataLabel: ({ label, percentage }) =>
          Text(`${percentage.toFixed(1)}%`, {
            style: new TextStyle({
              fontSize: 10,
              color: "#444444",
              fontFamily: "Noto Sans JP",
              fontWeight: "bold",
            }),
          }),
      },
    });

    export default function App() {
      return (
        <Widget
          width="auto"
          height="500px"
          widget={chart}
          renderer="svg"
        />
      );
    }
---

# Pie Chart Example

A pie chart displaying browser market share data with custom styled slices, data labels showing percentages, and a color-coded legend.
