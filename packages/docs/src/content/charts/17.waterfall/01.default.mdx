    import Widget from "@meursyphus/flitter-react";
    import { WaterfallChart } from "@meursyphus/headless-chart";
    import {
      Text,
      Container,
      EdgeInsets,
      TextStyle,
      Column,
      Row,
      SizedBox,
      MainAxisAlignment,
      MainAxisSize,
      BoxDecoration,
      Padding,
      Center,
    } from "@meursyphus/flitter";

    const data = {
      labels: ["Revenue", "COGS", "Gross Profit", "Marketing", "R&D", "Admin", "Net Income"],
      values: [500, -200, 0, -80, -120, -50, 0],
      totalIndices: [2, 6],
    };

    const chart = WaterfallChart({
      data,
      custom: {
        layout: ({ title, legends, plot }) =>
          Container({
            padding: EdgeInsets.only({ left: 55, bottom: 60, right: 20, top: 30 }),
            child: Column({
              children: [
                Center({
                  child: Text("Profit & Loss Breakdown ($K)", {
                    style: new TextStyle({
                      fontSize: 16,
                      fontWeight: "bold",
                      color: "#333333",
                      fontFamily: "Noto Sans JP",
                    }),
                  }),
                }),
                SizedBox({ height: 8 }),
                Row({
                  mainAxisAlignment: MainAxisAlignment.center,
                  children: [
                    Row({
                      mainAxisSize: MainAxisSize.min,
                      children: [
                        Container({ width: 12, height: 12, decoration: new BoxDecoration({ color: "#59a14f", borderRadius: 2 }) }),
                        SizedBox({ width: 4 }),
                        Text("Increase", { style: new TextStyle({ fontSize: 11, color: "#666", fontFamily: "Noto Sans JP" }) }),
                      ],
                    }),
                    SizedBox({ width: 16 }),
                    Row({
                      mainAxisSize: MainAxisSize.min,
                      children: [
                        Container({ width: 12, height: 12, decoration: new BoxDecoration({ color: "#e15759", borderRadius: 2 }) }),
                        SizedBox({ width: 4 }),
                        Text("Decrease", { style: new TextStyle({ fontSize: 11, color: "#666", fontFamily: "Noto Sans JP" }) }),
                      ],
                    }),
                    SizedBox({ width: 16 }),
                    Row({
                      mainAxisSize: MainAxisSize.min,
                      children: [
                        Container({ width: 12, height: 12, decoration: new BoxDecoration({ color: "#4e79a7", borderRadius: 2 }) }),
                        SizedBox({ width: 4 }),
                        Text("Total", { style: new TextStyle({ fontSize: 11, color: "#666", fontFamily: "Noto Sans JP" }) }),
                      ],
                    }),
                  ],
                }),
                SizedBox({ height: 12 }),
                plot,
              ],
            }),
          }),
        xAxisLabel: ({ name }) =>
          Padding({
            padding: EdgeInsets.only({ top: 4 }),
            child: Text(name, {
              style: new TextStyle({
                fontSize: 10,
                color: "#666666",
                fontFamily: "Noto Sans JP",
              }),
            }),
          }),
        yAxisLabel: ({ name }) =>
          Padding({
            padding: EdgeInsets.only({ right: 4 }),
            child: Text(name, {
              style: new TextStyle({
                fontSize: 11,
                color: "#666666",
                fontFamily: "Noto Sans JP",
              }),
            }),
          }),
        xAxisTick: () => Container({ height: 6, width: 1, color: "#DDDDDD" }),
        yAxisTick: () => Container({ height: 1, width: 6, color: "#DDDDDD" }),
        xAxisLine: () => Container({ color: "#BBBBBB", height: 1, width: Infinity }),
        yAxisLine: () => Container({ color: "#BBBBBB", width: 1, height: Infinity }),
        gridXLine: () => Container({ height: 1, color: "#F0F0F0" }),
        gridYLine: () => Container({ width: 1, color: "#F0F0F0" }),
      },
    });

    export default function App() {
      return (
        <Widget
          width="auto"
          height="450px"
          widget={chart}
          renderer="svg"
        />
      );
    }

# Waterfall Chart Example

A waterfall chart showing a profit and loss breakdown, with green bars for positive contributions (Revenue), red bars for expenses (COGS, Marketing, R&D, Admin), and blue bars for totals (Gross Profit, Net Income).
