  App.js: |
    import Widget from "@meursyphus/flitter-react";
    import { TreemapChart } from "@meursyphus/headless-chart";
    import {
      Text,
      Container,
      EdgeInsets,
      TextStyle,
      Column,
      Row,
      SizedBox,
      MainAxisAlignment,
      MainAxisSize,
      BoxDecoration,
      Padding,
      Center,
      Alignment,
    } from "@meursyphus/flitter";

    const colors = [
      "#4e79a7", "#f28e2b", "#e15759", "#76b7b2",
      "#59a14f", "#edc948", "#b07aa1", "#ff9da7",
    ];

    const data = {
      nodes: [
        { label: "Videos", value: 450 },
        { label: "Photos", value: 280 },
        { label: "Documents", value: 150 },
        { label: "Music", value: 120 },
        { label: "Apps", value: 200 },
        { label: "System", value: 180 },
        { label: "Downloads", value: 95 },
        { label: "Other", value: 60 },
      ],
      title: "Disk Space Usage (GB)",
    };

    const chart = TreemapChart({
      data,
      colors,
      custom: {
        layout: ({ title, legend, treemap }) =>
          Container({
            padding: EdgeInsets.all(20),
            child: Column({
              children: [
                Center({
                  child: Text("Disk Space Usage (GB)", {
                    style: new TextStyle({
                      fontSize: 16,
                      fontWeight: "bold",
                      color: "#333333",
                      fontFamily: "Noto Sans JP",
                    }),
                  }),
                }),
                SizedBox({ height: 12 }),
                treemap,
                SizedBox({ height: 12 }),
                legend,
              ],
            }),
          }),
        node: ({ label, value, color, ratio }) =>
          Container({
            width: Infinity,
            height: Infinity,
            padding: EdgeInsets.all(6),
            decoration: new BoxDecoration({
              color: color,
              border: BoxDecoration.border
                ? undefined
                : undefined,
            }),
            child: Column({
              mainAxisAlignment: MainAxisAlignment.center,
              children: [
                Text(label, {
                  style: new TextStyle({
                    fontSize: ratio > 0.1 ? 14 : 10,
                    fontWeight: "bold",
                    color: "#FFFFFF",
                    fontFamily: "Noto Sans JP",
                  }),
                }),
                SizedBox({ height: 2 }),
                Text(`${value} GB`, {
                  style: new TextStyle({
                    fontSize: ratio > 0.1 ? 11 : 8,
                    color: "rgba(255,255,255,0.85)",
                    fontFamily: "Noto Sans JP",
                  }),
                }),
              ],
            }),
          }),
        legendItem: ({ label, color }) =>
          Padding({
            padding: EdgeInsets.symmetric({ horizontal: 4, vertical: 2 }),
            child: Row({
              mainAxisSize: MainAxisSize.min,
              children: [
                Container({
                  width: 10,
                  height: 10,
                  decoration: new BoxDecoration({
                    color: color,
                    borderRadius: 2,
                  }),
                }),
                SizedBox({ width: 4 }),
                Text(label, {
                  style: new TextStyle({
                    fontSize: 11,
                    color: "#666666",
                    fontFamily: "Noto Sans JP",
                  }),
                }),
              ],
            }),
          }),
      },
    });

    export default function App() {
      return (
        <Widget
          width="auto"
          height="500px"
          widget={chart}
          renderer="svg"
        />
      );
    }
---

# Treemap Chart Example

A treemap chart visualizing disk space usage across different file categories. Each rectangle's size is proportional to the storage consumed.
