  App.js: |
    import Widget from "@meursyphus/flitter-react";
    import { FunnelChart } from "@meursyphus/headless-chart";
    import {
      Text,
      Container,
      EdgeInsets,
      TextStyle,
      Column,
      Row,
      SizedBox,
      MainAxisAlignment,
      MainAxisSize,
      BoxDecoration,
      Padding,
      Center,
    } from "@meursyphus/flitter";

    const data = {
      stages: [
        { label: "Website Visits", value: 15000, color: "#4e79a7" },
        { label: "Sign Ups", value: 8500, color: "#59a14f" },
        { label: "Trial Started", value: 4200, color: "#f28e2b" },
        { label: "Qualified Leads", value: 2100, color: "#edc948" },
        { label: "Purchases", value: 950, color: "#e15759" },
      ],
      title: "Sales Conversion Funnel",
    };

    const chart = FunnelChart({
      data,
      custom: {
        layout: ({ title, legends, funnel }) =>
          Container({
            padding: EdgeInsets.all(24),
            child: Column({
              children: [
                Center({
                  child: Text("Sales Conversion Funnel", {
                    style: new TextStyle({
                      fontSize: 18,
                      fontWeight: "bold",
                      color: "#333333",
                      fontFamily: "Noto Sans JP",
                    }),
                  }),
                }),
                SizedBox({ height: 8 }),
                Row({
                  mainAxisAlignment: MainAxisAlignment.center,
                  children: legends,
                }),
                SizedBox({ height: 16 }),
                funnel,
              ],
            }),
          }),
        legend: ({ label, color }) =>
          Padding({
            padding: EdgeInsets.symmetric({ horizontal: 6 }),
            child: Row({
              mainAxisSize: MainAxisSize.min,
              children: [
                Container({
                  width: 12,
                  height: 12,
                  decoration: new BoxDecoration({
                    color: color,
                    borderRadius: 2,
                  }),
                }),
                SizedBox({ width: 4 }),
                Text(label, {
                  style: new TextStyle({
                    fontSize: 11,
                    color: "#666666",
                    fontFamily: "Noto Sans JP",
                  }),
                }),
              ],
            }),
          }),
        dataLabel: ({ value, percentage, label }) =>
          Text(`${value.toLocaleString()} (${percentage.toFixed(1)}%)`, {
            style: new TextStyle({
              fontSize: 12,
              color: "#FFFFFF",
              fontWeight: "bold",
              fontFamily: "Noto Sans JP",
            }),
          }),
        stageLabel: ({ label }) =>
          Padding({
            padding: EdgeInsets.only({ right: 8 }),
            child: Text(label, {
              style: new TextStyle({
                fontSize: 13,
                color: "#444444",
                fontWeight: "bold",
                fontFamily: "Noto Sans JP",
              }),
            }),
          }),
      },
    });

    export default function App() {
      return (
        <Widget
          width="auto"
          height="450px"
          widget={chart}
          renderer="svg"
        />
      );
    }
---

# Funnel Chart Example

A funnel chart visualizing a sales conversion pipeline from website visits to purchases, with each stage showing the count and conversion percentage.
